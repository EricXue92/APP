**Goal:** Add a trilingual NTRP self-assessment quiz with 6 questions that calculates a recommended NTRP level. No auth required, no database changes.

**Architecture:** Hardcoded quiz questions with trilingual text in a service module. Two endpoints on a new router: GET questions and POST assess. Scoring sums option points and maps to an NTRP level.

**Tech Stack:** FastAPI, Pydantic v2, pytest + httpx

```python
from pydantic import BaseModel


class QuestionOption(BaseModel):
    key: str
    text: str
    points: int

    model_config = {"from_attributes": True}


class Question(BaseModel):
    id: int
    text: str
    options: list[QuestionOption]

    model_config = {"from_attributes": True}


class QuestionsResponse(BaseModel):
    questions: list[Question]

    model_config = {"from_attributes": True}


class AnswerItem(BaseModel):
    question_id: int
    selected: str


class AssessRequest(BaseModel):
    answers: list[AnswerItem]


class AssessResponse(BaseModel):
    ntrp_level: str
    ntrp_label: str

    model_config = {"from_attributes": True}
```

```python
NTRP_LABELS = {
    "2.0": {"zh-Hant": "2.0 初級", "zh-Hans": "2.0 初级", "en": "2.0 Beginner"},
    "2.5": {"zh-Hant": "2.5 初級", "zh-Hans": "2.5 初级", "en": "2.5 Beginner+"},
    "3.0": {"zh-Hant": "3.0 中初級", "zh-Hans": "3.0 中初级", "en": "3.0 Intermediate-Low"},
    "3.5": {"zh-Hant": "3.5 中級", "zh-Hans": "3.5 中级", "en": "3.5 Intermediate"},
    "4.0": {"zh-Hant": "4.0 中高級", "zh-Hans": "4.0 中高级", "en": "4.0 Intermediate-High"},
    "4.5": {"zh-Hant": "4.5 高級", "zh-Hans": "4.5 高级", "en": "4.5 Advanced"},
    "5.0": {"zh-Hant": "5.0 高級", "zh-Hans": "5.0 高级", "en": "5.0 Advanced+"},
}

SCORE_TO_NTRP = [
    (8, "2.0"),
    (10, "2.5"),
    (13, "3.0"),
    (16, "3.5"),
    (19, "4.0"),
    (22, "4.5"),
    (24, "5.0"),
]


def get_questions(lang: str) -> list[dict]:
    """Return quiz questions with text resolved to the given language."""
    result = []
    for q in QUESTIONS:
        result.append({
            "id": q["id"],
            "text": q["text"].get(lang, q["text"]["zh-Hant"]),
            "options": [
                {
                    "key": opt["key"],
                    "text": opt["text"].get(lang, opt["text"]["zh-Hant"]),
                    "points": opt["points"],
                }
                for opt in q["options"]
            ],
        })
    return result


def calculate_ntrp(answers: list[dict], lang: str) -> dict:
    """Validate answers, sum points, return NTRP level and localized label.

    Raises ValueError for missing questions or invalid option keys.
    """
    # Build lookup: question_id -> {key -> points}
    question_map = {}
    for q in QUESTIONS:
        option_map = {opt["key"]: opt["points"] for opt in q["options"]}
        question_map[q["id"]] = option_map

    # Validate all questions are answered
    answered_ids = {a["question_id"] for a in answers}
    expected_ids = {q["id"] for q in QUESTIONS}
    missing = expected_ids - answered_ids
    if missing:
        raise ValueError(f"Missing answers for question(s): {sorted(missing)}")

    # Validate option keys and sum points
    total = 0
    for a in answers:
        qid = a["question_id"]
        if qid not in question_map:
            raise ValueError(f"Unknown question_id: {qid}")
        option_map = question_map[qid]
        if a["selected"] not in option_map:
            raise ValueError(f"Invalid option '{a['selected']}' for question {qid}")
        total += option_map[a["selected"]]

    # Map total to NTRP level
    ntrp_level = "5.0"
    for max_points, level in SCORE_TO_NTRP:
        if total <= max_points:
            ntrp_level = level
            break

    ntrp_label = NTRP_LABELS[ntrp_level].get(lang, NTRP_LABELS[ntrp_level]["zh-Hant"])

    return {"ntrp_level": ntrp_level, "ntrp_label": ntrp_label}
```

```python
from fastapi import APIRouter, HTTPException, status

from app.dependencies import Lang
from app.schemas.ntrp_quiz import AssessRequest, AssessResponse, QuestionsResponse
from app.services.ntrp_quiz import calculate_ntrp, get_questions

router = APIRouter()


@router.get("/questions", response_model=QuestionsResponse)
async def get_quiz_questions(lang: Lang):
    questions = get_questions(lang)
    return {"questions": questions}


@router.post("/assess", response_model=AssessResponse)
async def assess_ntrp(body: AssessRequest, lang: Lang):
    try:
        answers = [{"question_id": a.question_id, "selected": a.selected} for a in body.answers]
        result = calculate_ntrp(answers, lang)
        return result
    except ValueError as e:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(e))
```
